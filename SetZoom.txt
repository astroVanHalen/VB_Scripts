' Macro Name: SetZoomAndResetView
' Author: astroVanHalen
' Created: 07/24/2025
' Modified: N/A
' Purpose: Prompts user for a zoom level (10-400), sets all sheets in the active workbook to that zoom, selects cell A1 on each sheet, and activates the first sheet
' Usage: Run via Ctrl + Z or Macros menu from any workbook; no specific selection required
' Notes: Validates zoom input; works from PERSONAL.XLSB or any workbook
'
Sub SetZoomAndResetView()
    Dim ws As Worksheet
    Dim zoomLevel As Variant
    
    ' Prompt for zoom level
    zoomLevel = InputBox("Zoom level? (numeric, 10-400):", "Set Zoom Level")
    
    ' Check for valid input or cancellation
    If zoomLevel = "" Then Exit Sub ' Exit if user cancels
    If Not IsNumeric(zoomLevel) Or zoomLevel < 10 Or zoomLevel > 400 Then
        MsgBox "Please enter a number between 10 and 400.", vbExclamation
        Exit Sub
    End If
    
    ' Set zoom, select A1, and activate first sheet
    Application.ScreenUpdating = False ' Improve performance
    For Each ws In ActiveWorkbook.Worksheets
        ws.Activate
        ws.Range("A1").Select
        ActiveWindow.Zoom = CLng(zoomLevel)
    Next ws
    ActiveWorkbook.Worksheets(1).Activate
    
    Application.ScreenUpdating = True
    MsgBox "Zoom level set to " & zoomLevel & "% and view reset to cell A1 on all sheets.", vbInformation
End Sub
